pip install django-crispy-forms==1.11.2
pip install Pillow==8.2.0
pip install whitenoise==2.0.6
pip install
pip install gunicorn==20.0.4
git clone https://github.com/Binshadcs/Event-Gateway.git
cd Event-Gateway
python manage.py runserver
